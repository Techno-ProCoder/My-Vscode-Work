 }
 if(sum==q)
 {

int fact(int r)
{
    int p=1;
    for(int i=1;i<=r;i++)
    {
        p=p*i;
    }
    return p;
}

        if(j>i)

            for(k=1;k<=n;k++)

    char comparisonstack1[50];  // All brackets
    char comparisonstack2[50];  // Reversed second half

    cout << "Enter the number of characters (including symbols and parentheses): ";
    cin >> m;

    if (m > 50) {
        cout << "Input too long!";
        return 1;
    }

    cout << "Enter the expression: ";
    for (i = 0; i < m; i++) {
        cin >> inputstack[i];
    }

    // Step 1: Push only brackets into comparisonstack1
    for (i = 0; i < m; i++) {
        if (inputstack[i] == '(' || inputstack[i] == '{' || inputstack[i] == '[' ||
            inputstack[i] == ')' || inputstack[i] == '}' || inputstack[i] == ']') {
            comparisonstack1[top++] = inputstack[i];
        }
    }

    // Step 2: If count is odd, it's unbalanced
    if (top % 2 != 0) {
        cout << "Expression has no balanced parentheses." << endl;
        return 0;
    }

    // Step 3: Copy second half into comparisonstack2 in reverse order
    for (i = top - 1; i >= top / 2; i--) {
        comparisonstack2[top2++] = comparisonstack1[i];
    }

    // Step 4: Compare pairs
    for (i = 0; i < top / 2; i++) {
        char a = comparisonstack1[i];
        char b = comparisonstack2[i];

        if (!((a == '(' && b == ')') || 
              (a == '{' && b == '}') || 
              (a == '[' && b == ']'))) {
            flag = 1;
            break;
        }
    }

    // Final Output
    if (flag == 0)
        cout << "Expression has balanced parentheses." << endl;
    else
        cout << "Expression has no balanced parentheses." << endl;

    return 0;
}

// First Missing Positive
#include <iostream>

int firstMissingPositive(int nums[], int n) 
{
    // Step 1: Place each number at its correct index
    for (int i = 0; i < n; i++) 
{
        while (nums[i] >= 1 && nums[i] <= n && nums[nums[i] - 1] != nums[i]) 
{
            swap(nums[i], nums[nums[i] - 1]);
}

}

    // Step 2: Find first missing positive
    for (int i = 0; i < n; i++) 
{
        if (nums[i] != i + 1)
            return i + 1;
}

    return n + 1;
}

// Sort Even-Placed In Increasing And Odd-Placed In Decreasing Order
#include <iostream>
using namespace std;
/* Partition function for ascending Quick Sort */
int partitionAsc(int arr[], int low, int high) 
{
    int pivot = arr[high];
    int i = low - 1;
    for (int j = low; j < high; j++) 
    {
        if (arr[j] < pivot) 
        {
            i++;
            swap(arr[i], arr[j]);
        }
    }
    swap(arr[i + 1], arr[high]);
    return i + 1;
}
/* Quick Sort in ascending order */
void quickSortAsc(int arr[], int low, int high) 
{
    if (low < high) 
    {
        int pi = partitionAsc(arr, low, high);
        quickSortAsc(arr, low, pi - 1);
        quickSortAsc(arr, pi + 1, high);
    }
}
/* Partition function for descending Quick Sort */
int partitionDesc(int arr[], int low, int high) 
{
    int pivot = arr[high];
    int i = low - 1;
    for (int j = low; j < high; j++) 
    {
        if (arr[j] > pivot) 
        {
            i++;
            swap(arr[i], arr[j]);
        }
    }
    swap(arr[i + 1], arr[high]);
    return i + 1;
}
/* Quick Sort in descending order */
void quickSortDesc(int arr[], int low, int high) 
{
    if (low < high) 
    {
        int pi = partitionDesc(arr, low, high);
        quickSortDesc(arr, low, pi - 1);
        quickSortDesc(arr, pi + 1, high);
    }
}
/* In-place separation of even-index and odd-index elements */
int separateEvenOddIndex(int arr[], int n) 
{
    int evenPos = 0;
    for (int i = 0; i < n; i++) 
    {
        if (i % 2 == 0) 
        {
            swap(arr[i], arr[evenPos]);
            evenPos++;
        }
    }
    return evenPos; // count of even-index elements
}

    // Step 3: Sort odd-index elements in descending order

    cout << arr[i] << " ";
}

// Power Of Three
def isPowerOfThree(n: int) -> bool:
    if n <= 0:
        return False
    if n == 1:
        return True
    if n % 3 != 0:
        return False
    return isPowerOfThree(n // 3)

// Given two strings ransomNote and magazine, return true if ransomNote can be constructed by using the letters from magazine and false otherwise.
from collections import Counter

def canConstruct(ransomNote, magazine):
    return Counter(ransomNote) <= Counter(magazine)

// Valid Anagram
def isAnagram(s: str, t: str) -> bool:
    if len(s) != len(t):
        return False
    return sorted(s) == sorted(t)


// Given a non-empty array of integers nums, every element appears twice except for one. Find that single one.
def singleNumber(nums):
    result = 0
    for num in nums:
        result ^= num
    return result

// Max Consecutive Ones
int findMaxConsecutiveOnes(vector<int>& nums) 
{
    int count = 0;
    int count1 = 0;
    for (int i = 0; i < nums.size(); i++) 
{
        if (nums[i] == 1)
            count++;
        else
            count = 0;
        count1 = max(count1, count);
}
    return count1;
}

// Diameter Of Binary Tree
int diameter = 0;
int depth(TreeNode* root) 
{
    if (!root) 
      return 0;
    int left = depth(root->left);
    int right = depth(root->right);
    diameter = max(diameter, left + right);
    return 1 + max(left, right);
}

// Tribonacci Series
//(Recursive Code)
#include <iostream>
using namespace std;
int trib(int n)
{
    if (n == 1) return 0;          
    if (n == 2 || n == 3) return 1; 
    return trib(n - 1) + trib(n - 2) + trib(n - 3); 
}

    cout << trib(n);   
}
//(Iterative Code)
#include <iostream>
using namespace std;
int trib(int n)
{
    if (n == 1) 
    return 0;
    if (n == 2 || n == 3) 
    return 1;
    int t1 = 0; 
    int t2 = 1; 
    int t3 = 1;
    int t;
    for (int i = 4; i <= n; i++)
    {
        t = t1 + t2 + t3;
        t1 = t2;
        t2 = t3;
        t3 = t;
    }
    return t3;
}
